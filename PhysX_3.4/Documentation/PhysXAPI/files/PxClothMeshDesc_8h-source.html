00132 PX_INLINE bool PxClothMeshDesc::isValid() const
00133 {
00134     if(points.count < 3)    //at least 1 trig's worth of points
00135         return false;
00136     if(points.count > 0xffff && flags & PxMeshFlag::e16_BIT_INDICES)
00137         return false;
00138     if(!points.data)
00139         return false;
00140     if(points.stride < sizeof(PxVec3))  //should be at least one point's worth of data
00141         return false;
00142 
00143     if(invMasses.data && invMasses.stride < sizeof(float))
00144         return false;
00145     if(invMasses.data && invMasses.count != points.count)
00146         return false;
00147 
00148     if (!triangles.count && !quads.count)   // no support for non-indexed mesh
00149         return false;
00150     if (triangles.count && !triangles.data)
00151         return false;
00152     if (quads.count && !quads.data)
00153         return false;
00154 
00155     PxU32 indexSize = (flags & PxMeshFlag::e16_BIT_INDICES) ? sizeof(PxU16) : sizeof(PxU32);
00156     if(triangles.count && triangles.stride < indexSize*3) 
00157         return false; 
00158     if(quads.count && quads.stride < indexSize*4)
00159         return false;
00160 
00161     return true;
00162 }
