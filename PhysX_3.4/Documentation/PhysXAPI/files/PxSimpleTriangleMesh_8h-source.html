00139 PX_INLINE bool PxSimpleTriangleMesh::isValid() const
00140 {
00141     // Check geometry
00142     if(points.count > 0xffff && flags & PxMeshFlag::e16_BIT_INDICES)
00143         return false;
00144     if(!points.data)
00145         return false;
00146     if(points.stride < sizeof(PxVec3))  //should be at least one point's worth of data
00147         return false;
00148 
00149     // Check topology
00150     // The triangles pointer is not mandatory
00151     if(triangles.data)
00152     {
00153         // Indexed mesh
00154         PxU32 limit = (flags & PxMeshFlag::e16_BIT_INDICES) ? sizeof(PxU16)*3 : sizeof(PxU32)*3;
00155         if(triangles.stride < limit) 
00156             return false; 
00157     }
00158     return true;
00159 }
