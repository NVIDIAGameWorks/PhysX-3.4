00043 template <typename storageType, storageType bitMask>
00044 class PxBitAndDataT
00045 {
00046   public:
00047     PX_FORCE_INLINE PxBitAndDataT(const PxEMPTY)
00048     {
00049     }
00050     PX_FORCE_INLINE PxBitAndDataT() : mData(0)
00051     {
00052     }
00053     PX_FORCE_INLINE PxBitAndDataT(storageType data, bool bit = false)
00054     {
00055         mData = bit ? storageType(data | bitMask) : data;
00056     }
00057 
00058     PX_CUDA_CALLABLE PX_FORCE_INLINE operator storageType() const
00059     {
00060         return storageType(mData & ~bitMask);
00061     }
00062     PX_CUDA_CALLABLE PX_FORCE_INLINE void setBit()
00063     {
00064         mData |= bitMask;
00065     }
00066     PX_CUDA_CALLABLE PX_FORCE_INLINE void clearBit()
00067     {
00068         mData &= ~bitMask;
00069     }
00070     PX_CUDA_CALLABLE PX_FORCE_INLINE storageType isBitSet() const
00071     {
00072         return storageType(mData & bitMask);
00073     }
00074 
00075   protected:
00076     storageType mData;
00077 };
